{-# OPTIONS_GHC -Wno-orphans #-}
module Proarrow.Promonad (
  Promonad(..),
  Procomonad(..)
) where

import Proarrow.Core (Profunctor, Promonad(..), type (~>), Ob)

class Profunctor p => Procomonad p where
  extract :: p a b -> (a ~> b)
  duplicate :: p a c -> (forall b. Ob b => (p a b, p b c) -> r) -> r
